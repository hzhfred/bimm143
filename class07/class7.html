library(ggplot2)

df <- as.data.frame(pca$x)

df$samples <- colnames(rna.data) 
df$condition <- substr(colnames(rna.data),1,2)

p <- ggplot(df) + 
        aes(PC1, PC2, label=samples, col=condition) + 
        geom_label(show.legend = FALSE)
p + labs(title="PCA of RNASeq Data",
